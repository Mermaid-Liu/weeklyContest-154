var game = function(guess, answer) {
   let map=[]
   let ans=0
   map.push(answer[0]-guess[0])
    map.push(answer[1]-guess[1])
    map.push(answer[2]-guess[2])
    console.log(map)
    for(let item of map){
        if(item===0){
            ans+=1
        }
    }
    return ans
};

# 第二题

var fraction = function(cont) {
    //递推
        var answer=[]
        var len=cont.length
        var i=0
        //边界条件
        if(len!==1){
            var temp=fraction(cont.slice(i+1,len))
            console.log(temp)
            answer[0]=cont[i]*temp[0]+temp[1]
            answer[1]=temp[0]
            return answer
        }else{
            return  [cont[i],1]
        }
};
